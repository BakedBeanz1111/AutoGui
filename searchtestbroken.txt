public void searchTest(string name,string windowname)
        {
            Condition apptitle = new PropertyCondition(AutomationElement.NameProperty, "Untitled - Notepad");
            AutomationElement window = AutomationElement.RootElement.FindFirst(TreeScope.Children, apptitle);
            window.SetFocus();

            Condition menutitle = new PropertyCondition(AutomationElement.NameProperty, "Edit");
            AutomationElement editbar = window.FindFirst(TreeScope.Children, menutitle);
            editbar.SetFocus();
            
            
            var isInvokable = (bool) editbar.GetCurrentPropertyValue(AutomationElement.IsInvokePatternAvailableProperty);
            if (isInvokable)
            {
                var invokePattern = editbar.GetCurrentPattern(InvokePattern.Pattern) as InvokePattern;
                invokePattern.Invoke();
            }
            else
            {
                System.Windows.Point p = editbar.GetClickablePoint();
                System.Windows.Forms.Cursor.Position = new System.Drawing.Point((int)p.X, (int)p.Y);
                Mouse.Click();
            }
            
        }
